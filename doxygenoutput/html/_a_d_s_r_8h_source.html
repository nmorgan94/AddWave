78 inline float ADSR::process() {
79  switch (state) {
80  case env_idle:
81  break;
82  case env_attack:
83  output = attackBase + output * attackCoef;
84  if (output >= 1.0) {
85  output = 1.0;
86  state = env_decay;
87  }
88  break;
89  case env_decay:
90  output = decayBase + output * decayCoef;
91  if (output <= sustainLevel) {
92  output = sustainLevel;
93  state = env_sustain;
94  }
95  break;
96  case env_sustain:
97  break;
98  case env_release:
99  output = releaseBase + output * releaseCoef;
100  if (output <= 0.0) {
101  output = 0.0;
102  state = env_idle;
103  }
104  }
105  return output;
106 }
